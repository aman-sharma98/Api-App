CREATE DATABASE audios;
use audios;
CREATE TABLE song(id INT NOT NULL UNIQUE AUTO_INCREMENT, name VARCHAR(100) NOT NULL, duration INT UNSIGNED NOT NULL, upload_time DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL, PRIMARY KEY(id));
CREATE TABLE podcast(id INT NOT NULL UNIQUE AUTO_INCREMENT, name VARCHAR(100) NOT NULL, duration INT UNSIGNED NOT NULL, upload_time DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL, host VARCHAR(100) NOT NULL, PRIMARY KEY(id));
CREATE TABLE participants(id INT NOT NULL UNIQUE, participant_1 VARCHAR(100), participant_2 VARCHAR(100), participant_3 VARCHAR(100), participant_4 VARCHAR(100), participant_5 VARCHAR(100), participant_6 VARCHAR(100), participant_7 VARCHAR(100), participant_8 VARCHAR(100), participant_9 VARCHAR(100), participant_10 VARCHAR(100), FOREIGN KEY(id) REFERENCES podcast(id));
CREATE TABLE audiobook(id INT NOT NULL UNIQUE AUTO_INCREMENT, title VARCHAR(100) NOT NULL, author VARCHAR(100) NOT NULL, narrator VARCHAR(100) NOT NULL, duration INT UNSIGNED NOT NULL, upload_time DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL, PRIMARY KEY(id));
